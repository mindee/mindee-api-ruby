def initialize(prediction_class, page)
  @id = page['id'].to_i
  @prediction = prediction_class.new(page['prediction'], @id)
  @orientation = Orientation.new(page['orientation'], @id)
end

def to_s
  out_str = String.new
  title = "Page #{@id}"
  out_str << "#{title}\n"
  out_str << ('-' * title.size)
  out_str << "\n#{@prediction}"
end
