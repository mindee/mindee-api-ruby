def to_s
  out_str = String.new
  out_str << "#{@account_number}; " if @account_number
  out_str << "#{@iban}; " if @iban
  out_str << "#{@routing_number}; " if @routing_number
  out_str << "#{@swift}; " if @swift
  out_str.strip
end
