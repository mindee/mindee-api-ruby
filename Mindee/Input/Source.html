def self.convert_to_unicode_escape(string)
  unicode_escape_string = ''.dup
  string.each_char do |char|
    unicode_escape_string << if char.bytesize > 1
                               "\\u#{char.unpack1('U').to_s(16).rjust(4, '0')}"
                             else
                               char
                             end
  end
  unicode_escape_string
end
