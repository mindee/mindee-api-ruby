def self.universal_object?(str_dict)
  common_keys = [
    'value',
    'polygon',
    'rectangle',
    'page_id',
    'confidence',
    'quadrangle',
    'values',
    'raw_value',
  ]
  str_dict.each_key { |key| return true unless common_keys.include?(key.to_s) }
  false
end
