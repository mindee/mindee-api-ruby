def initialize(http_error, url, code)
  @status_code = code
  @api_code = http_error['code']
  @api_details = http_error['details']
  @api_message = http_error['message']
  super("#{url} #{@status_code} HTTP error: #{@api_details} - #{@api_message}")
end
