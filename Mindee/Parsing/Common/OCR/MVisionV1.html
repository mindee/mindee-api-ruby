def reconstruct_vertically(coordinates, page_id, x_margin)
  line_arr = OCRLine.new([])
  @pages[page_id].all_lines.each do |line|
    line.each do |word|
      line_arr.push(word) if Geometry.below?(word.polygon, coordinates, x_margin / 2, x_margin * 2)
    end
  end
  line_arr
end

def to_s
  out_str = String.new
  @pages.map do |page|
    out_str << "\n"
    out_str << page.to_s
  end
  out_str.strip
end
