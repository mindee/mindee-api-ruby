def initialize(prediction, page_id)
  @cropping = []
  prediction['cropping'].each do |item|
    @cropping.push(PositionField.new(item, page_id))
  end
  super()
end

def to_s
  cropping = @cropping.join("\n #{' ' * 18}")
  out_str = String.new
  out_str << "\n:Document Cropper: #{cropping}".rstrip
  out_str
end
