def to_table_line
  printable = table_printable_values
  out_str = String.new
  out_str << format('| %- 16s', printable[:contract_type])
  out_str << format('| %- 11s', printable[:department])
  out_str << format('| %- 37s', printable[:description])
  out_str << format('| %- 26s', printable[:employer])
